151 /* find where and when a fast-moving Earth satellite, op, will rise and set and
152  * it's transit circumstances. all times are mjd, angles rads e of n.
153  * dis is the angle down from the local topo horizon, in rads (see riset()).
154  * idea is to walk forward in time looking for alt+dis==0 crossings.
155  * initial time step is a few degrees (based on average daily motion).
156  * we stop as soon as we see both a rise and set.
157  * N.B. we assume *np and *op are working copies we can mess up.
158  */
159 static void
160 e_riset_cir (Now *np, Obj *op, double dis, RiseSet *rp)
161 {
162 #define DEGSTEP 2 /* time step is about this many degrees */
163  int steps; /* max number of time steps */
164  double dt; /* time change per step, days */
165  double t0, t1; /* current and next mjd values */
166  double a0, a1; /* altitude at t0 and t1 */
167  int rise, set; /* flags to check when we find these events */
168  int i;
169 
170  dt = DEGSTEP * (1.0/360.0/op->es_n);
171  steps = (int)(1.0/dt);
172  rise = set = 0;
173  rp->rs_flags = 0;
174 
175  if (obj_cir (np, op) < 0) {
176  rp->rs_flags |= RS_ERROR;
177  return;
178  }
179 
180  t0 = mjd;
181  a0 = op->s_alt + dis;
182 
183  for (i = 0; i < steps && (!rise || !set); i++) {
184  mjd = t1 = t0 + dt;
185  if (obj_cir (np, op) < 0) {
186  rp->rs_flags |= RS_ERROR;
187  return;
188  }
189  a1 = op->s_alt + dis;
190 
191  if (a0 < 0 && a1 > 0 && !rise) {
192  /* found a rise event -- interate to refine */
193  switch (find_0alt (10./3600., 5./SPD, dis, np, op)) {
194  case 0: /* ok */
195  rp->rs_risetm = np->n_mjd;
196  rp->rs_riseaz = op->s_az;
197  rise = 1;
198  break;
199  case -1: /* obj_cir error */
200  rp->rs_flags |= RS_RISERR;
201  return;
202  case -2: /* converged but not today */ /* FALLTHRU */
203  case -3: /* probably never up */
204  rp->rs_flags |= RS_NORISE;
205  return;
206  }
207  } else if (a0 > 0 && a1 < 0 && !set) {
208  /* found a setting event -- interate to refine */
209  switch (find_0alt (10./3600., 5./SPD, dis, np, op)) {
210  case 0: /* ok */
211  rp->rs_settm = np->n_mjd;
212  rp->rs_setaz = op->s_az;
213  set = 1;
214  break;
215  case -1: /* obj_cir error */
216  rp->rs_flags |= RS_SETERR;
217  return;
218  case -2: /* converged but not today */ /* FALLTHRU */
219  case -3: /* probably circumpolar */
220  rp->rs_flags |= RS_NOSET;
221  return;
222  }
223  }
224 
225  t0 = t1;
226  a0 = a1;
227  }
228 
229  /* instead of transit, for satellites we find time of maximum
230  * altitude, if we know both the rise and set times.
231  */
232  if (rise && set) {
233  double tt, al, az;
234  if (find_maxalt (np, op, rp->rs_risetm, rp->rs_settm, &tt, &al, &az) < 0) {
235  rp->rs_flags |= RS_TRANSERR;
236  return;
237  }
238  rp->rs_trantm = tt;
239  rp->rs_tranalt = al;
240  rp->rs_tranaz = az;
241  } else
242  rp->rs_flags |= RS_NOTRANS;
243 
244  /* check for some bad conditions */
245  if (!rise) {
246  if (a0 > 0)
247  rp->rs_flags |= RS_CIRCUMPOLAR;
248  else
249  rp->rs_flags |= RS_NORISE;
250  }
251  if (!set) {
252  if (a0 < 0)
253  rp->rs_flags |= RS_NEVERUP;
254  else
255  rp->rs_flags |= RS_NOSET;
256  }
257 }

259 /* given a Now at noon and a dt from np, in hours, for a first approximation
260  * to a rise or set event, refine the event by searching for when alt+dis = 0.
261  * return 0: if find one within 12 hours of noon with np and op set to the
262  * better time and circumstances;
263  * return -1: if error from obj_cir;
264  * return -2: if converges but not today;
265  * return -3: if does not converge at all (probably circumpolar or never up);
266  */
267 static int
268 find_0alt (
269 double dt, /* hours from initial np to first guess at event */
270 double fstep, /* first step size, days */
271 double dis, /* horizon displacement, rads */
272 Now *np, /* working Now -- starts with mjd is noon, returns as answer */
273 Obj *op) /* working object -- returns as answer */
274 {
275 #define TMACC (0.01/SPD) /* convergence accuracy, days; tight for stable az */
276 #define MAXPASSES 20 /* max iterations to try */
277 #define MAXSTEP (12.0/24.0) /* max time step,days (to detect flat)*/
278 
279  double a0 = 0;
280  double mjdn = mjd;
281  int npasses;
282 
283  /* insure initial guess is today -- if not, move by 24 hours */
284  if (dt < -12.0 && !find_0alt (dt+24, fstep, dis, np, op))
285  return (0);
286  mjd = mjdn;
287  if (dt > 12.0 && !find_0alt (dt-24, fstep, dis, np, op))
288  return (0);
289  mjd = mjdn;
290 
291  /* convert dt to days for remainder of algorithm */
292  dt /= 24.0;
293 
294  /* use secant method to look for s_alt + dis == 0 */
295  npasses = 0;
296  do {
297  double a1;
298 
299  mjd += dt;
300  if (obj_cir (np, op) < 0)
301  return (-1);
302  a1 = op->s_alt;
303 
304  dt = (npasses == 0) ? fstep : (dis+a1)*dt/(a0-a1);
305  a0 = a1;
306 
307  if (++npasses > MAXPASSES || fabs(dt) >= MAXSTEP)
308  return (-3);
309 
310  } while (fabs(dt)>TMACC);
311  // fprintf (stderr, "%s 0alt npasses = %d\n", op->o_name, npasses);
312 
313  /* return codes */
314  return (fabs(mjdn-mjd) < .5 ? 0 : -2);
315 
316 #undef MAXPASSES
317 #undef MAXSTEP
318 #undef TMACC
319 }

321 /* find when the given object transits. start the search when LST matches the
322  * object's RA at noon.
323  * if ok, return 0 with np and op set to the transit conditions; if can't
324  * converge return -1; if converges ok but not today return -2.
325  * N.B. we assume np is passed set to local noon.
326  */
327 static int
328 find_transit (double dt, Now *np, Obj *op)
329 {
330 #define MAXLOOPS 10
331 #define MAXERR (1./3600.) /* hours */
332  double mjdn = mjd;
333  double lst;
334  int i;
335 
336  /* insure initial guess is today -- if not, move by 24 hours */
337  if (dt < -12.0)
338  dt += 24.0;
339  if (dt > 12.0)
340  dt -= 24.0;
341 
342  i = 0;
343  do {
344  mjd += dt/24.0;
345  if (obj_cir (np, op) < 0)
346  return (-1);
347  now_lst (np, &lst);
348  dt = (radhr(op->s_gaera) - lst);
349  if (dt < -12.0)
350  dt += 24.0;
351  if (dt > 12.0)
352  dt -= 24.0;
353  } while (++i < MAXLOOPS && fabs(dt) > MAXERR);
354 
355  /* fprintf (stderr, "%s find_transit loops = %d, dt = %g seconds\n", op->o_name, i, dt*3600); */
356 
357  /* return codes */
358  if (i == MAXLOOPS)
359  return (-1);
360  return (fabs(mjd - mjdn) < 0.5 ? 0 : -2);
361 
362 #undef MAXLOOPS
363 #undef MAXERR
364 }

366 /* find the mjd time of max altitude between the given rise and set times.
367  * N.B. we assume *np and *op are working copies we can modify.
368  * return 0 if ok, else -1.
369  */
370 static int
371 find_maxalt (
372 Now *np,
373 Obj *op,
374 double tr, double ts, /* mjd of rise and set */
375 double *tp, /* time of max altitude */
376 double *alp, double *azp) /* max altitude and transit az at said time */
377 {
378 #define MAXLOOPS 100 /* max loops */
379 #define MAXERR (1.0/SPD) /* days */
380 
381  double l, r; /* times known to bracket max alt */
382  double m1, m2; /* intermediate range points inside l and r */
383  double a1, a2; /* alt at m1 and m2 */
384  int nloops; /* max loop check */
385 
386  /* want rise before set */
387  while (ts < tr)
388  tr -= 1.0/op->es_n;
389 
390  /* init time bracket */
391  l = tr;
392  r = ts;
393 
394  /* ternary search for max */
395  for (nloops = 0; r - l > MAXERR && nloops < MAXLOOPS; nloops++) {
396 
397  mjd = m1 = (2*l + r)/3;
398  obj_cir (np, op);
399  a1 = op->s_alt;
400 
401  mjd = m2 = (l + 2*r)/3;
402  obj_cir (np, op);
403  a2 = op->s_alt;
404 
405  if (a1 < a2)
406  l = m1;
407  else
408  r = m2;
409  }
410  // fprintf (stderr, "tern nloops = %d\n", nloops);
411  if (nloops >= MAXLOOPS)
412  return (-1);
413 
414  /* best is between l and r */
415  mjd = *tp = (l+r)/2;
416  obj_cir (np, op);
417  *alp = op->s_alt;
418  *azp = op->s_az;
419 
420  return (0);
421 #undef MAXERR
422 #undef MAXLOOPS
423 }

463 /* Solve Kepler's equation */
464 /* Inputs: */
465 /* MeanAnomaly Time Since last perigee, in radians. */
466 /* PI2 = one complete orbit. */
467 /* Eccentricity Eccentricity of orbit's ellipse. */
468 /* Output: */
469 /* TrueAnomaly Angle between perigee, geocenter, and */
470 /* current position. */
471 
472 static
473 double Kepler(double MeanAnomaly, double Eccentricity)
474 {
475 register double E; /* Eccentric Anomaly */
476 register double Error;
477 register double TrueAnomaly;
478 
479  E = MeanAnomaly ;/*+ Eccentricity*sin(MeanAnomaly); -- Initial guess */
480  do
481  {
482  Error = (E - Eccentricity*sin(E) - MeanAnomaly)
483  / (1 - Eccentricity*cos(E));
484  E -= Error;
485  }
486  while (ABS(Error) >= Epsilon);
487 
488  if (ABS(E-PI) < Epsilon)
489  TrueAnomaly = PI;
490  else
491  TrueAnomaly = 2*atan(sqrt((1+Eccentricity)/(1-Eccentricity))
492  *tan(E/2));
493  if (TrueAnomaly < 0)
494  TrueAnomaly += PI2;
495 
496  return TrueAnomaly;
497 }

500 GetSubSatPoint(double SatX, double SatY, double SatZ, double T,
501 double *Latitude, double *Longitude, double *Height)
502 {
503  double r;
504  /* ECD: long i; */
505 
506  r = sqrt(SQR(SatX) + SQR(SatY) + SQR(SatZ));
507 
508  *Longitude = PI2*((T-SidDay)*SiderealSolar + SidReference)
509  - atan2(SatY,SatX);
510 
511  /* ECD:
512  * want Longitude in range -PI to PI , +W
513  */
514  range (Longitude, 2*PI);
515  if (*Longitude > PI)
516  *Longitude -= 2*PI;
517 
518  *Latitude = atan(SatZ/sqrt(SQR(SatX) + SQR(SatY)));
519 
520 #define SSPELLIPSE
521 #ifdef SSPELLIPSE
522  /* ECD */
523  *Height = r - EarthRadius*(sqrt(1-(2*EarthFlat-SQR(EarthFlat))*SQR(sin(*Latitude))));
524 #else
525  *Height = r - EarthRadius;
526 #endif

548 
549 static void
550 GetSatPosition(double EpochTime, double EpochRAAN, double EpochArgPerigee,
551 double SemiMajorAxis, double Inclination, double Eccentricity,
552 double RAANPrecession, double PerigeePrecession, double T,
553 double TrueAnomaly, double *X, double *Y, double *Z, double *Radius,
554 double *VX, double *VY, double *VZ)
555 
556 {
557  double RAAN,ArgPerigee;
558 
559 
560  double Xw,Yw,VXw,VYw; /* In orbital plane */
561  double Tmp;
562  double Px,Qx,Py,Qy,Pz,Qz; /* Escobal transformation 31 */
563  double CosArgPerigee,SinArgPerigee;
564  double CosRAAN,SinRAAN,CoSinclination,SinInclination;
565 
566  *Radius = SemiMajorAxis*(1-SQR(Eccentricity))
567  / (1+Eccentricity*cos(TrueAnomaly));
568 
569  if (*Radius <= EarthRadius)
570  *Radius = EarthRadius;
571 
572 
573  Xw = *Radius * cos(TrueAnomaly);
574  Yw = *Radius * sin(TrueAnomaly);
575 
576  Tmp = sqrt(GM/(SemiMajorAxis*(1-SQR(Eccentricity))));
577 
578  VXw = -Tmp*sin(TrueAnomaly);
579  VYw = Tmp*(cos(TrueAnomaly) + Eccentricity);
580 
581  ArgPerigee = EpochArgPerigee + (T-EpochTime)*PerigeePrecession;
582  RAAN = EpochRAAN - (T-EpochTime)*RAANPrecession;
583 
584  CosRAAN = cos(RAAN); SinRAAN = sin(RAAN);
585  CosArgPerigee = cos(ArgPerigee); SinArgPerigee = sin(ArgPerigee);
586  CoSinclination = cos(Inclination); SinInclination = sin(Inclination);

589  Py = CosArgPerigee*SinRAAN + SinArgPerigee*CosRAAN*CoSinclination;
590  Pz = SinArgPerigee*SinInclination;
591  Qx = -SinArgPerigee*CosRAAN - CosArgPerigee*SinRAAN*CoSinclination;
592  Qy = -SinArgPerigee*SinRAAN + CosArgPerigee*CosRAAN*CoSinclination;
593  Qz = CosArgPerigee*SinInclination;

596  *Y = Py*Xw + Qy*Yw;
597  *Z = Pz*Xw + Qz*Yw;

600  *VY = Py*VXw + Qy*VYw;

604 /* Compute the site postion and velocity in the RA based coordinate
605  system. SiteMatrix is set to a matrix which is used by GetTopoCentric
606  to convert geocentric coordinates to topocentric (observer-centered)
607  coordinates. */
608 
609 static void
610 GetSitPosition(double SiteLat, double SiteLong, double SiteElevation,
611 double CrntTime, double *SiteX, double *SiteY, double *SiteZ, double *SiteVX,
612 double *SiteVY, MAT3x3 SiteMatrix)
613 {
614  static double G1,G2; /* Used to correct for flattening of the Earth */
615  static double CosLat,SinLat;
616  static double OldSiteLat = -100000; /* Used to avoid unneccesary recomputation */
617  static double OldSiteElevation = -100000;
618  double Lat;
619  double SiteRA; /* Right Ascension of site */
620  double CosRA,SinRA;
621 
622  if ((SiteLat != OldSiteLat) || (SiteElevation != OldSiteElevation))
623  {
624  OldSiteLat = SiteLat;
625  OldSiteElevation = SiteElevation;
626  Lat = atan(1/(1-SQR(EarthFlat))*tan(SiteLat));
627 
628  CosLat = cos(Lat);
629  SinLat = sin(Lat);
630 
631  G1 = EarthRadius/(sqrt(1-(2*EarthFlat-SQR(EarthFlat))*SQR(SinLat)));
632  G2 = G1*SQR(1-EarthFlat);
633  G1 += SiteElevation;
634  G2 += SiteElevation;
635  }
636 
637 
638  SiteRA = PI2*((CrntTime-SidDay)*SiderealSolar + SidReference)
639  - SiteLong;
640  CosRA = cos(SiteRA);
641  SinRA = sin(SiteRA);
642 
643 
644  *SiteX = G1*CosLat*CosRA;
645  *SiteY = G1*CosLat*SinRA;
646  *SiteZ = G2*SinLat;
647  *SiteVX = -SidRate * *SiteY;
648  *SiteVY = SidRate * *SiteX;
649 
650  SiteMatrix[0][0] = SinLat*CosRA;
651  SiteMatrix[0][1] = SinLat*SinRA;
652  SiteMatrix[0][2] = -CosLat;
653  SiteMatrix[1][0] = -SinRA;
654  SiteMatrix[1][1] = CosRA;
655  SiteMatrix[1][2] = 0.0;
656  SiteMatrix[2][0] = CosRA*CosLat;
657  SiteMatrix[2][1] = SinRA*CosLat;
658  SiteMatrix[2][2] = SinLat;
659 }

661 static void
662 GetRange(double SiteX, double SiteY, double SiteZ, double SiteVX,
663 double SiteVY, double SatX, double SatY, double SatZ, double SatVX,
664 double SatVY, double SatVZ, double *Range, double *RangeRate)
665 {
666  double DX,DY,DZ;
667 
668  DX = SatX - SiteX; DY = SatY - SiteY; DZ = SatZ - SiteZ;
669 
670  *Range = sqrt(SQR(DX)+SQR(DY)+SQR(DZ));
671 
672  *RangeRate = ((SatVX-SiteVX)*DX + (SatVY-SiteVY)*DY + SatVZ*DZ)
673  / *Range;
674 }

677  (observer centered) coordinates */
678 
679 static void
680 GetTopocentric(double SatX, double SatY, double SatZ, double SiteX,
681 double SiteY, double SiteZ, MAT3x3 SiteMatrix, double *X, double *Y,
682 double *Z)
683 {
684  SatX -= SiteX;
685  SatY -= SiteY;
686  SatZ -= SiteZ;
687 
688  *X = SiteMatrix[0][0]*SatX + SiteMatrix[0][1]*SatY
689  + SiteMatrix[0][2]*SatZ;
690  *Y = SiteMatrix[1][0]*SatX + SiteMatrix[1][1]*SatY
691  + SiteMatrix[1][2]*SatZ;
692  *Z = SiteMatrix[2][0]*SatX + SiteMatrix[2][1]*SatY
693  + SiteMatrix[2][2]*SatZ;

741  Formulas are from "Explanatory Supplement to the Astronomical Ephemeris".
742  Also init the sidereal reference */
743 
744 static void
745 InitOrbitRoutines(double EpochDay, int AtEod)
746 {
747  double T,T2,T3,Omega;
748  int n;
749  double SunTrueAnomaly,SunDistance;
750 
751  T = (floor(EpochDay)-0.5)/36525;
752  T2 = T*T;
753  T3 = T2*T;
754 
755  SidDay = floor(EpochDay);
756 
757  SidReference = (6.6460656 + 2400.051262*T + 0.00002581*T2)/24;
758  SidReference -= floor(SidReference);
759 
760  /* Omega is used to correct for the nutation and the abberation */
761  Omega = AtEod ? (259.18 - 1934.142*T) * RadiansPerDegree : 0.0;
762  n = (int)(Omega / PI2);
763  Omega -= n*PI2;
764 
765  SunEpochTime = EpochDay;
766  SunRAAN = 0;
767 
768  SunInclination = (23.452294 - 0.0130125*T - 0.00000164*T2
769  + 0.000000503*T3 +0.00256*cos(Omega)) * RadiansPerDegree;
770  SunEccentricity = (0.01675104 - 0.00004180*T - 0.000000126*T2);
771  SunArgPerigee = (281.220833 + 1.719175*T + 0.0004527*T2
772  + 0.0000033*T3) * RadiansPerDegree;
773  SunMeanAnomaly = (358.475845 + 35999.04975*T - 0.00015*T2
774  - 0.00000333333*T3) * RadiansPerDegree;
775  n = (int)(SunMeanAnomaly / PI2);
776  SunMeanAnomaly -= n*PI2;
777 
778  SunMeanMotion = 1/(365.24219879 - 0.00000614*T);
779 
780  SunTrueAnomaly = Kepler(SunMeanAnomaly,SunEccentricity);
781  SunDistance = SunSemiMajorAxis*(1-SQR(SunEccentricity))
782  / (1+SunEccentricity*cos(SunTrueAnomaly));
783 
784  SinPenumbra = (SunRadius-EarthRadius)/SunDistance;
785  CosPenumbra = sqrt(1-SQR(SinPenumbra));
786 }
